```typescript components/MioButtons.tsx
'use client';

import { useState } from 'react';
import { useMio } from '@mio-xyz/sdk/react';

export function MioButtons() {
  const { connect, handleMioCallback, getContext, isLoading } = useMio();
  const [accessToken, setAccessToken] = useState<string | null>(null);

  const connectUser = async () => {
    if (accessToken) return;
    await connect();
  };

  const finishCallback = async () => {
    const tokens = await handleMioCallback();
    setAccessToken(tokens.accessToken);
  };

  const ask = async () => {
    if (!accessToken) return;
    await getContext({ query: 'Tell me about the userâ€™s professional history', accessToken });
  };

  return (
    <div className="flex gap-2">
      <button onClick={connectUser} disabled={isLoading}>Connect</button>
      <button onClick={finishCallback}>Handle callback</button>
      <button onClick={ask} disabled={!accessToken}>Ask Mio</button>
    </div>
  );
}
```
